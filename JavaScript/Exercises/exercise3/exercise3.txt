<!-- BONUS -->

let n1 = Number(prompt("Please enter the first number: "));
let n2 = Number(prompt("Please enter the second number: "));

let operation = Number(prompt("Please enter the operation you want to perform (1 for addition, 2 for subtration, 3 for multiplication, 4 for division): " ));

let result;

switch (operation) {
  case 1:
    result = n1 + n2;
    break;

  case 2: 
    result = n1 - n2;
    break;

  case 3:
    result = n1 * n2;
    break;

  case 4: 
    result = n1 / n2;
    break;
}

alert("The result is " + result);
